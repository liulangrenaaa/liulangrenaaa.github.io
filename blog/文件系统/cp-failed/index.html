#!/bin/bash

touch a

f() {
        let i=0
        while true; do
                let i++
                rm -f b
                strace -o /tmp/cp${BASHPID}.trace cp a b || break
                echo $i
        done
}

cleanup() {
        kill -9 %1 %2
}

f &
f &

trap cleanup exit

wait
